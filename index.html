const REPORT_SHEET = '신고접수'; const REWARD_SHEET = '포상관리'; const ADMIN_EMAIL = 'thinkvision17@gmail.com'; function doGet(e) { try { const action = e.parameter.action || 'list'; if (action === 'list') return res(getList()); if (action === 'stats') return res(getStats()); if (action === 'submit') return res(submitReport(e.parameter)); if (action === 'confirm') return res(confirmReport(e.parameter)); if (action === 'reject') return res(rejectReport(e.parameter)); return res({ ok: false, msg: 'Unknown action' }); } catch (err) { return res({ ok: false, msg: err.toString() }); } } // JSON 응답 헬퍼 function res(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ── 신고 목록 조회 ── function getList() { const ws = SpreadsheetApp.openById(SHEET_ID).getSheetByName(REPORT_SHEET); const data = ws.getDataRange().getValues(); if (data.length <= 1) return { ok: true, data: [] }; } // ── 신고 제출 ── function submitReport(p) { const ws = SpreadsheetApp.openById(SHEET_ID).getSheetByName(REPORT_SHEET); const lastRow = ws.getLastRow(); const id = 'RPT-' + String(lastRow).padStart(4, '0'); const now = Utilities.formatDate(new Date(), 'Asia/Seoul', 'yyyy-MM-dd HH:mm'); // 사진 Google Drive 저장 let photoUrl = ''; try { if (p.photo && p.photo.startsWith('data:image')) { const parts = p.photo.split(','); const mimeType = parts[0].match(/:(.*?);/)[1]; const blob = Utilities.newBlob(Utilities.base64Decode(parts[1]), mimeType, id + '.jpg'); const folder = DriveApp.getFoldersByName('주민신고_사진').hasNext() ? DriveApp.getFoldersByName('주민신고_사진').next() : DriveApp.createFolder('주민신고_사진'); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); photoUrl = 'https://drive.google.com/uc?id=' + file.getId(); } } catch(e) { photoUrl = '사진저장실패:' + e.message; } ws.appendRow([ id, now, p.type || '', p.plate || '', p.location || '', photoUrl, p.detail || '', p.contact || '', '접수대기', '', '', '', p.reward || '', '미지급', '' ]); MailApp.sendEmail({ to: ADMIN_EMAIL, subject: `[주민신고접수] ${p.type} — ${p.plate}`, body: `새 신고가 접수되었습니다. ■ 신고 ID : ${id} ■ 접수 일시: ${now} ■ 신고 유형: ${p.type} ■ 위반 차량: ${p.plate} ■ 위반 장소: ${p.location || '미입력'} ■ 상세 내용: ${p.detail || '미입력'} ■ 연락처 : ${p.contact || '미입력 (익명)'} ■ 포상 선택: ${p.reward || '미선택'} Google Sheets에서 처리 상태를 업데이트해 주세요.` }); return { ok: true, id }; } // ── 위반 확정 ── function confirmReport(p) { const ws = SpreadsheetApp.openById(SHEET_ID).getSheetByName(REPORT_SHEET); const data = ws.getDataRange().getValues(); const idx = data.findIndex(r => r[0] === p.id); const sheetRow = idx + 1; ws.getRange(sheetRow, 9).setValue('처리완료'); ws.getRange(sheetRow, 10).setValue(now); ws.getRange(sheetRow, 11).setValue('위반확정'); ws.getRange(sheetRow, 12).setValue(code); // 포상관리 시트 기록 const rws = SpreadsheetApp.openById(SHEET_ID).getSheetByName(REWARD_SHEET); const month = Utilities.formatDate(new Date(), 'Asia/Seoul', 'yyyy-MM'); rws.appendRow([data[idx][7], p.id, data[idx][12], 10000, month, '', '', '']); // 결과 이메일 const contact = data[idx][7]; const reward = data[idx][12]; if (contact && contact.includes('@')) { MailApp.sendEmail({ to: contact, subject: '[신고처리결과] 신고하신 건이 처리되었습니다', body: `안녕하세요. 은어송하늘채리버뷰 관리사무소입니다. 접수하신 신고가 아래와 같이 처리되었습니다. ───────────────────────────────────── ■ 신고 유형 : ${data[idx][2]} ■ 위반 차량 : ${data[idx][3]} ■ 처리 결과 : 위반 사실 확정 ───────────────────────────────────── 포상 수령 안내 ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 포상 인증번호 : ${code} ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 포상 수령을 원하시면 관리사무소 방문 시 위 인증번호를 직원에게 제시해 주시기 바랍니다. ※ 포상: ${reward || '미선택'} (월 최대 10,000원) ※ 유효기간: 당월 말일까지 은어송하늘채리버뷰 관리사무소 드림` }); } return { ok: true, code }; } // ── 미확정 처리 ── function rejectReport(p) { const ws = SpreadsheetApp.openById(SHEET_ID).getSheetByName(REPORT_SHEET); const data = ws.getDataRange().getValues(); const idx = data.findIndex(r => r[0] === p.id); if (idx < 0) return { ok: false, msg: '신고를 찾을 수 없습니다.' }; const now = Utilities.formatDate(new Date(), 'Asia/Seoul', 'yyyy-MM-dd HH:mm'); const sheetRow = idx + 1; ws.getRange(sheetRow, 9).setValue('미확정'); ws.getRange(sheetRow, 10).setValue(now); ws.getRange(sheetRow, 11).setValue('위반미확정'); const contact = data[idx][7]; if (contact && contact.includes('@')) { MailApp.sendEmail({ to: contact, subject: '[신고처리결과] 신고하신 건 처리 결과 안내', body: `안녕하세요. 은어송하늘채리버뷰 관리사무소입니다. 접수하신 신고를 현장 확인 및 CCTV 영상 검토한 결과, 위반 사실을 확인하기 어려워 신고를 종결 처리하였습니다. ───────────────────────────────────── ■ 신고 유형 : ${data[idx][2]} ■ 위반 차량 : ${data[idx][3]} ■ 처리 결과 : 위반 사실 미확인 — 신고 종결 ───────────────────────────────────── 추후 동일 차량의 재위반 발견 시 재신고해 주시기 바랍니다. 신고에 협조해 주셔서 감사합니다. 은어송하늘채리버뷰 관리사무소 드림` }); } return { ok: true }; }
